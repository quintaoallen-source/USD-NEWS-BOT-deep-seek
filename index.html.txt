    const impactColor  = (v) => v === "high" ? "#ef4444" : v === "medium" ? "#f59e0b" : "#22c55e";
    const tagColor     = (t) => ({ FED:"#6366f1", DATA:"#0ea5e9", DXY:"#8b5cf6", GEOPOLITICAL:"#f97316", GOLD:"#eab308", PREDICTION:"#ec4899" }[t] || "#64748b");

    const expectColor  = (e) => ({ beat:"#22c55e", inline:"#94a3b8", miss:"#ef4444", unknown:"#f59e0b" }[e] || "#64748b");
    const reactionCol  = (r) => r === "bullish" ? "#22c55e" : r === "bearish" ? "#ef4444" : "#94a3b8";
    const reactionIcon = (r) => r === "bullish" ? "▲" : r === "bearish" ? "▼" : "◆";

    async function callClaude(prompt) {
      const res = await fetch('/api/news', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ prompt }),
      });
      if (!res.ok) {
        const err = await res.json();
        throw new Error(err.error || `HTTP ${res.status}`);
      }
      return res.json();
    }

    function USDNewsBot() {
      const [active, setActive] = useState("all");
      const [news, setNews] = useState(null);
      const [loading, setLoading] = useState(false);
      const [error, setError] = useState(null);
      const [lastFetched, setLastFetched] = useState(null);
      const [autoRefresh, setAutoRefresh] = useState(false);
      const intervalRef = useRef(null);

      const fetchNews = async (cat = active) => {
        setLoading(true);
        setError(null);
        setNews(null);
        try {
          const result = await callClaude(PROMPTS[cat]);
          setNews(result);
          setLastFetched(new Date());
        } catch (e) {
          console.error(e);
          setError(e.message || "Failed to fetch. Please try again.");
        } finally {
          setLoading(false);
        }
      };

      const switchCat = (cat) => {
        setActive(cat);
        setNews(null);
        setError(null);
      };

      useEffect(() => {
        clearInterval(intervalRef.current);
        if (autoRefresh) {
          intervalRef.current = setInterval(() => fetchNews(active), 5 * 60 * 1000);
        }
        return () => clearInterval(intervalRef.current);
      }, [autoRefresh, active]);

      const sent = news ? sentCfg(news.sentiment) : null;
      const cat = CATEGORIES.find(c => c.id === active);

      return (

                <div style={{ width:38, height:38, background:"linear-gradient(135deg,#6366f1,#8b5cf6)", borderRadius:10, display:"flex", alignItems:"center", justifyContent:"center", fontSize:18 }}>📡</div>

            )}

            {CATEGORIES.map(c => (
              <button key={c.id} onClick={() => switchCat(c.id)} style={{
                background: active===c.id ? "linear-gradient(135deg,#6366f1,#8b5cf6)" : "#0f1929",
                border: active===c.id ? "none" : "1px solid #1a2740",

                {c.emoji} {c.label}
              </button>
            ))}

              </div>
            </div>
          )}
          <style>{`* { box-sizing:border-box; } ::-webkit-scrollbar { display:none; }`}</style>
        </div>
      );
    }
